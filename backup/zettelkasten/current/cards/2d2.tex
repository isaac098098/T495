\zheader{Caracterización de identificaciones}{caracterización, identificación}{2d2}

\begin{theorem}
Si $f : X \longrightarrow Y$ es una función, son equivalentes
\begin{enumerate}[label=\textnormal{(\roman*)}]
\item $f$ es identificación.
\item $U$ es abierto en $Y$ si y sólo si $f^{-1}(U)$ es abierto en $X$.
\item $F$ es cerrado en $Y$ si y sólo si $f^{-1}(F)$ es cerrado en $X$.
\end{enumerate}
\end{theorem}

\begin{proof}
({\romannumeral 1}) $\implies$ ({\romannumeral 2}).  Si $f$ es identificación entonces $f$ es, en particular, continua, y por tanto $U$ abierto en $Y$ implica $f^{-1}(U)$ abierto en $X$. Supogase ahora que $f^{-1}(U)$ es abierto en $X$ con $U \subset Y$. Entonces $U$ es abierto en $X$ por definición de topología de identificación. Como $U$ fue arbitrario se tiene el resultado.
\bigskip

({\romannumeral 2}) $\implies$ ({\romannumeral 3}). Se tiene que
\begin{align*}
    F \text{ es cerrado en } Y & \iff X - F \text{ es abierto en } Y \\
                               & \iff f^{-1}(X - F) = Y-f^{-1}(F) && \text{ es abierto en } X, \\
                               & && \text{ por hipótesis} \\
                               & \iff f^{-1}(F) \text{ es cerrado en } X.
\end{align*}

({\romannumeral 3}) $\implies$ ({\romannumeral 2}). Es similar al punto anterior.
\bigskip

({\romannumeral 2}) $\implies$ ({\romannumeral 1}). Sea $\mathcal{T}$ la topología de $Y$. Si se verifica ({\romannumeral 2}), entonces la $\mathcal{T}$ hace continua a $f$. Más aún, si hay otra topología $\mathcal{T'}$ que hace continua a $f$, entonces $U \in \mathcal{T}$ implica que $f^{-1}(U)$ es abierto en $X$, y por tanto $U \in \mathcal{T}$ por hipótesis. Luego $\mathcal{T'} \subset \mathcal{T}$ y como $\mathcal{T'}$ fue arbitraria, entonces $\mathcal{T}$ es de hecho más fina en $Y$ que cualquier otra que haga continua a $f$. Es fácil verificar que sólo existe una topología sobre $Y$ con esta propiedad y es la topología de identificación. Luego, $\mathcal{T}$ es la topología de identificación coinducida por $f$, es decir, $f$ es una identificación.
\end{proof}
